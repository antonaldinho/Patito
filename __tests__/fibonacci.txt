program fibonacci;
var
	int a, b;
	int c, n, i;

main ()
{
	a = 0;
	b = 1;
    read(n);
    if (n<0) {
        print("Incorrect Input");
    } else {
        if (n == 0) {
            print(a);
        } else {
            if (n == 1) {
                print(b);
            } else {
                from (i=2) until (i==n) do {
                    c = a + b;
                    a = b;
                    b = c;
                    i = i + 1;
                }
                print(b);
            }
        }
    }
}
